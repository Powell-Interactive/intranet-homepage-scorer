function showPanel(anchor, color) {
  const title = anchor.dataset.title;
  const body  = anchor.dataset.body;
  const owner = anchor.dataset.owner;
  panel.style.borderTopColor = COLORS[color] || '#083C49';
  panel.innerHTML = `

  `;

  const dot = anchor._dot;
  const r   = dot.getBoundingClientRect();
  const vw  = document.documentElement.clientWidth;
  const vh  = document.documentElement.clientHeight;
  const pw  = 270;

  panel.style.display = 'block';
  panel.style.pointerEvents = 'auto';
  const ph = panel.offsetHeight || 200;

  let left = r.left + r.width/2 - pw/2;
  left = Math.max(8, Math.min(left, vw - pw - 8));

  let top = r.bottom + 8;
  if (top + ph > vh - 8) top = r.top - ph - 8;
  top = Math.max(8, top);

  panel.style.left = left + 'px';
  panel.style.top  = top  + 'px';

  requestAnimationFrame(() => {
    panel.style.opacity = '1';
    panel.style.transform = 'translateY(0) scale(1)';
  });
}

function hidePanel() {
  panel.style.opacity = '0';
  panel.style.transform = 'translateY(6px) scale(.97)';
  setTimeout(() => {
    panel.style.display = 'none';
    panel.style.pointerEvents = 'none';
  }, 180);
  activeAnchor = null;
}

function animateCount(el, target, duration) {
  const start = performance.now();
  function step(now) {
    const progress = Math.min((now - start) / duration, 1);
    const eased = 1 - Math.pow(1 - progress, 3);
    el.textContent = Math.round(eased * target);
    if (progress < 1) requestAnimationFrame(step);
  }
  requestAnimationFrame(step);
}

  dl.href = src;
  dl.download = label.replace(/[^a-z0-9]/gi, '-').toLowerCase() + '.png';
  lb.style.display = 'flex';
  requestAnimationFrame(() => lb.style.opacity = '1');
}
function closeLightbox() {
  const lb = document.getElementById('lightbox');
  lb.style.display = 'none';
}
document.addEventListener('keydown', e => { if (e.key === 'Escape') closeLightbox(); });
